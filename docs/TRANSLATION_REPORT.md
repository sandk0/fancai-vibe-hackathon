"""
Automated documentation translation script
"""
import os
import re
from pathlib import Path

def translate_file(input_file, output_file, translator_api):
    """
    Translate markdown file preserving structure

    Args:
        input_file: Path to English markdown file
        output_file: Path to Russian markdown file
        translator_api: Translation API (DeepL, GPT, etc.)
    """
    with open(input_file, 'r', encoding='utf-8') as f:
        content = f.read()

    # Extract code blocks to preserve them
    code_blocks = []
    def save_code_block(match):
        code_blocks.append(match.group(0))
        return f"CODE_BLOCK_{len(code_blocks)-1}"

    # Temporarily replace code blocks
    content = re.sub(r'```.*?```', save_code_block, content, flags=re.DOTALL)

    # Translate content (using API)
    translated = translator_api.translate(content, target_lang='RU')

    # Restore code blocks
    for i, block in enumerate(code_blocks):
        # Translate comments in code blocks
        block = translate_code_comments(block)
        translated = translated.replace(f"CODE_BLOCK_{i}", block)

    # Write translated file
    with open(output_file, 'w', encoding='utf-8') as f:
        f.write(translated)

    print(f"‚úÖ Translated: {input_file} ‚Üí {output_file}")

def translate_code_comments(code_block):
    """Translate only comments in code blocks"""
    # Python comments
    code_block = re.sub(
        r'#\s*(.+)$',
        lambda m: f"# {translate_text(m.group(1))}",
        code_block,
        flags=re.MULTILINE
    )
    # JavaScript/TypeScript comments
    code_block = re.sub(
        r'//\s*(.+)$',
        lambda m: f"// {translate_text(m.group(1))}",
        code_block,
        flags=re.MULTILINE
    )
    return code_block
